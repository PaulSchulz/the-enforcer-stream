\recette{Banana Cake}
\preptime{10 min} \baketime{180\textdegree C/350\textdegree F, 35 min} \cake{10}

\recipe{
        \unit[1]{cup}   & Self-raising flour \\
        \unit[1]{cup}   & Caster sugar \\
        \unit[50]{g}    & Butter, melted \\
        1               & Egg, lightly beaten \\
        \unit[1]{tsp}   & Vanilla essence \\
        \unit[1/4]{cup} & Milk \\
        1               & Banana, mashed \\
}{
    \item Preheat oven to 180\textdegree C/160\textdegree C fan-forced.
    Line a 7cm deep, 10x20cm (base measurement) loaf pan with baking paper.
    
    \item Sift flour into a large bowl. Stir in sugar.
    Make a well in the center. Add butter, egg, vanilla, milk and banana.
    Stir until just combined.
    
    \item Pour into prepared pan. Bake for 30-35 minutes
    or until a skewer inserted into the center comes out clean.

    \item Set aside in pan for 10 minutes to cool slightly.
    Serve warm or transfer to wire rack to cool completely.
}

Optional:  Cinnamon can also be added to taste.

\info{Bananas are not native to Ukraine.}
